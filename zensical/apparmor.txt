#include <tunables/global>

profile hassio_zensical flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay (official HA example pattern)
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Zensical CLI and pip-installed binaries (Alpine pip → /usr/local/bin/)
  /usr/local/bin/** ix,

  # Nginx runtime dirs (tmp, logs)
  /var/lib/nginx/{,**} rwk,

  # Zensical helper scripts
  /opt/zensical/bin/* ix,

  # HA add-on data and mounts
  /config/{,**} rw,
  /addon_config/{,**} rw,

  # Network (nginx ingress, DNS)
  network inet  stream,
  network inet  dgram,
  network inet6 stream,
  network inet6 dgram,
  network unix  stream,
  network unix  dgram,
}
