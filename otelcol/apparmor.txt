#include <tunables/global>

profile otelcol flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # ── s6-overlay init system ───────────────────────────────────────────
  /command/**                     mrix,
  /package/**                     mr,
  /run/s6/                        r,
  /run/s6/**                      rwlk,
  /run/s6-linux-init-container-results/   r,
  /run/s6-linux-init-container-results/** rw,

  # ── bashio + helpers ────────────────────────────────────────────────
  /usr/lib/bashio/**              mr,
  /usr/bin/jq                     mrix,
  /usr/bin/curl                   mrix,
  /usr/bin/wget                   mrix,

  # ── otelcol-contrib binary ──────────────────────────────────────────
  /usr/bin/otelcol-contrib        mrix,

  # ── Python 3 (ha-bridge) ────────────────────────────────────────────
  /usr/bin/python3*               mrix,
  /usr/local/bin/python3*         mrix,
  /usr/lib/python3*/              r,
  /usr/lib/python3*/**            mr,
  /usr/local/lib/python3*/        r,
  /usr/local/lib/python3*/**      mr,

  # ── System libraries (musl / Alpine) ────────────────────────────────
  /lib/                           r,
  /lib/**                         mr,
  /usr/lib/                       r,
  /usr/lib/**                     mr,
  /usr/local/lib/                 r,
  /usr/local/lib/**               mr,

  # ── System config & certs ───────────────────────────────────────────
  /etc/                           r,
  /etc/**                         r,

  # ── otelcol config files ────────────────────────────────────────────
  /etc/otelcol-contrib/           r,
  /etc/otelcol-contrib/**         r,
  /tmp/                           r,
  /tmp/**                         rw,

  # ── HA mounts ───────────────────────────────────────────────────────
  /config/                        r,
  /config/**                      r,
  /ssl/                           r,
  /ssl/**                         r,
  /addon_config/                  r,
  /addon_config/**                rw,
  /share/                         r,
  /share/**                       rw,

  # ── proc / sys (otelcol reads for self-metrics, Python for runtime) ─
  /proc/                          r,
  /proc/**                        r,
  owner /proc/**                  rw,
  /sys/                           r,
  /sys/**                         r,

  # ── Device nodes ────────────────────────────────────────────────────
  /dev/null                       rw,
  /dev/urandom                    r,
  /dev/random                     r,
  /dev/tty                        rw,
  /dev/pts/**                     rw,

  # ── Runtime sockets & pipes (excluding Docker) ──────────────────────
  /run/                           r,
  /run/**                         rw,
  /var/run/                       r,
  /var/run/**                     rw,

  # Explicitly deny the Docker socket — we use the Supervisor API instead
  deny /var/run/docker.sock       rwklx,
  deny /run/docker.sock           rwklx,

  # ── Networking ──────────────────────────────────────────────────────
  network inet  stream,
  network inet  dgram,
  network inet6 stream,
  network inet6 dgram,
  network unix  stream,
  network unix  dgram,

  # ── Capabilities (s6-overlay needs chown/setuid/kill at startup) ────
  capability chown,
  capability dac_override,
  capability fowner,
  capability fsetid,
  capability kill,
  capability net_bind_service,
  capability setgid,
  capability setuid,
  capability sys_chroot,

  # Deny broad system capabilities
  deny capability sys_admin,
  deny capability sys_ptrace,
  deny capability net_admin,
  deny capability sys_rawio,
  deny capability mknod,
}
