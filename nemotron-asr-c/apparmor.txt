#include <tunables/global>

profile ha_addon_nemotron_asr_c flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Debian system binaries and libraries
  /bin/**           mrix,
  /usr/bin/**       mrix,
  /lib/**           mr,
  /lib64/**         mr,
  /usr/lib/**       mr,
  /usr/local/bin/** mrix,
  /usr/local/lib/** mr,  # includes libnemotron_asr.so, PyTorch, numpy

  # s6-overlay v3 (nemotron-asr-c also has a down script)
  /command/**  mrix,
  /etc/s6-overlay/**        r,
  /etc/s6-overlay/**/run    ix,
  /etc/s6-overlay/**/finish ix,
  /etc/s6-overlay/**/down   ix,
  /run/s6/** rw,
  /run/s6-linux-init-container-results/** rw,

  # Boot-time .nemo -> .bin converter tools (run script: cd /usr/src/app)
  /usr/src/app/ r,
  /usr/src/app/** r,

  # Persistent data: .nemo model, converted weights, HF cache (all in /data)
  /data/**  rw,
  /tmp/**   rw,

  # /proc: needed by Python and PyTorch
  /proc/ r,
  /proc/** r,

  capability kill,
  capability dac_override,

  # Wyoming TCP (listen 10370) + HTTPS for model download (huggingface_hub) + DNS
  network inet tcp,
  network inet6 tcp,
  network inet udp,
  network inet6 udp,
  network unix,

  deny capability sys_admin,
  deny capability sys_module,
  deny capability sys_rawio,
  deny capability net_admin,
  deny capability sys_ptrace,
  deny /proc/sysrq-trigger rw,
  deny /proc/*/mem rw,
  deny /proc/kcore r,
  deny /sys/firmware/** rw,
}
