#include <tunables/global>

# Full-whitelist profile — no blanket `file,` rule. AppArmor is the outer
# boundary for the whole container; the inner transcribe/converter split
# is enforced by unix users (DAC). Every path below maps to a service:
# s6 machinery, the Wyoming server (transcribe), the conversion worker
# (converter), and the root oneshots (init-perms, discovery).
profile hassio_transcribe_cpp flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Privilege handling: s6-setuidgid drops to transcribe/converter,
  # init-perms chowns /data subtrees, s6 signals services.
  capability chown,
  capability fowner,
  capability fsetid,
  capability setuid,
  capability setgid,
  capability kill,
  capability dac_override,
  capability dac_read_search,
  signal (send,receive) set=(kill,term,int,hup,cont,usr1,usr2),

  # S6-Overlay (official HA example pattern)
  /init rix,
  /bin/** rix,
  /usr/bin/** rix,
  /run/{s6,s6-rc*,service}/** rix,
  /package/** rmix,
  /command/** rmix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/s6-overlay/** rix,
  /run/ rwk,
  /run/** rwk,
  /dev/tty rw,
  /dev/console rw,
  /dev/pts/** rw,
  # joblib/torch shared-memory segments (converter venv)
  /dev/shm/ rw,
  /dev/shm/** rwk,

  # Bashio (discovery + service run scripts)
  /usr/lib/bashio/** rix,
  /tmp/ rwk,
  /tmp/** rwk,

  # Read-only system config (ssl certs, passwd/group for setuidgid,
  # resolv.conf/hosts for DNS)
  /etc/** r,

  # Python runtime + our package/deps + libtranscribe/ggml/quantize tool
  /usr/lib/python3*/** r,
  /usr/lib/python3*/lib-dynload/** mr,
  /usr/local/bin/* rix,
  /usr/local/lib/** mr,
  /usr/local/share/transcribe-cpp/** r,
  /usr/share/** r,

  # ggml/torch CPU feature probing
  @{PROC}/cpuinfo r,
  @{PROC}/meminfo r,
  @{PROC}/stat r,
  @{PROC}/@{pid}/** r,
  /sys/devices/system/cpu/** r,
  /sys/devices/system/node/** r,

  # Persistent data: models (transcribe), custom GGUFs + per-family
  # conversion venvs (converter). Venvs execute python symlinks and
  # pip-installed native wheels -> m + ix there only.
  /data/ rw,
  /data/** rwk,
  /data/convert-venv/** rwmkix,

  # Worker IPC socket (the only channel between the two users)
  /run/transcribe-cpp/ rw,
  /run/transcribe-cpp/** rwk,

  # Network: Wyoming on 10380/tcp, HF Hub + pip downloads, DNS;
  # unix stream for the convert socket; netlink for getaddrinfo.
  network inet  stream,
  network inet  dgram,
  network inet6 stream,
  network inet6 dgram,
  network unix  stream,
  network unix  dgram,
  network netlink raw,
}
