  #include <abstractions/base>

  # Debian system binaries and libraries
  /bin/**           mrix,
  /usr/bin/**       mrix,
  /lib/**           mr,
  /lib64/**         mr,
  /usr/lib/**       mr,
  /usr/local/bin/** mrix,
  /usr/local/lib/** mr,

  # s6-overlay v3
  /command/**  mrix,
  /etc/s6-overlay/**      r,
  /etc/s6-overlay/**/run    ix,
  /etc/s6-overlay/**/finish ix,
  /run/s6/** rw,
  /run/s6-linux-init-container-results/** rw,

  /data/**  rw,

  # /proc: needed by Python and the MNN runtime

  /proc/** r,

  network inet tcp,
  network inet6 tcp,
  network inet udp,

  network unix,

  deny capability sys_module,
  deny capability sys_rawio,
  deny capability net_admin,
  deny capability sys_ptrace,
  deny /proc/sysrq-trigger rw,
  deny /proc/*/mem rw,
  deny /proc/kcore r,
  deny /sys/firmware/** rw,
}
