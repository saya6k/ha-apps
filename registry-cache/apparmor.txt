#include <tunables/global>

profile hassio_registry_cache flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Signals
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /sbin/** ix,
  /usr/sbin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** mr,
  /command/** ix,

  # registry + nginx binaries
  /usr/bin/registry ix,
  /usr/sbin/nginx ix,

  # Bundled templates + registry-definitions (read-only)
  /etc/nginx/** r,
  /etc/registry/** r,
  /etc/registry-cache/** r,

  # Rendered config + unix sockets
  /run/registry-cache/ rwk,
  /run/registry-cache/** rwk,

  # Registry blob/repository storage
  /data/registry-cache/ rwk,
  /data/registry-cache/** rwk,

  # nginx runtime temp dirs
  /var/lib/nginx/ rwk,
  /var/lib/nginx/** rwk,
  /var/log/nginx/ rwk,
  /var/log/nginx/** rwk,

  # Runtime / proc / sys
  /run/ rwk,
  /run/** rwk,
  /proc/ r,
  /proc/** r,
  owner /proc/*/fd/** rw,
  /sys/** r,

  # Devices
  /dev/null rw,
  /dev/urandom r,
  /dev/tty rw,
  /dev/pts/** rw,

  # Explicit deny — this app never needs the host Docker socket
  deny /run/docker.sock rwklx,
  deny /var/run/docker.sock rwklx,

  # Network
  network inet stream,
  network inet6 stream,
  network unix stream,

  capability chown,
  capability dac_override,
  capability fowner,
  capability fsetid,
  capability kill,
  capability setgid,
  capability setuid,
  capability sys_chroot,

  deny capability sys_admin,
  deny capability sys_ptrace,
  deny capability net_admin,
  deny capability sys_rawio,
  deny capability mknod,
}
